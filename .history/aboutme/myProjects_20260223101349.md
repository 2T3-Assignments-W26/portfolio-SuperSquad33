```C#
 static void GenerateMaze()
 {
     maze = new char[height, width];
     for (int y = 0; y < height; y++)
         for (int x = 0; x < width; x++)
             maze[y, x] = '#';

     // Random start
     start = (rand.Next(width), rand.Next(height));
     CarveMaze(start.x, start.y);

     // Find all valid edge positions for exit (not surrounded by walls)
     List<(int x, int y)> validExits = new List<(int x, int y)>();
     for (int x = 0; x < width; x++)
     {
         if (IsValidExit(x, 0)) validExits.Add((x, 0));
         if (IsValidExit(x, height - 1)) validExits.Add((x, height - 1));
     }
     for (int y = 1; y < height - 1; y++)
     {
         if (IsValidExit(0, y)) validExits.Add((0, y));
         if (IsValidExit(width - 1, y)) validExits.Add((width - 1, y));
     }

     // Remove start if it's on the edge
     validExits.RemoveAll(e => e == start);

     // Pick a random valid exit
     if (validExits.Count == 0)
     {
         // fallback: just pick a random edge cell (should be rare)
         exit = (0, 0);
     }
     else
     {
         exit = validExits[rand.Next(validExits.Count)];
     }

     maze[start.y, start.x] = 'S';
     maze[exit.y, exit.x] = 'E';
     player = start;
 }
 ```
