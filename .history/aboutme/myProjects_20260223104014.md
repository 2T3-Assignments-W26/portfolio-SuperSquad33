```C#
 static void GenerateMaze()
 {
     maze = new char[height, width];
     for (int y = 0; y < height; y++)
         for (int x = 0; x < width; x++)
             maze[y, x] = '#';

     // Random start
     start = (rand.Next(width), rand.Next(height));
     CarveMaze(start.x, start.y);

     // Find all valid edge positions for exit (not surrounded by walls)
     List<(int x, int y)> validExits = new List<(int x, int y)>();
     for (int x = 0; x < width; x++)
     {
         if (IsValidExit(x, 0)) validExits.Add((x, 0));
         if (IsValidExit(x, height - 1)) validExits.Add((x, height - 1));
     }
     for (int y = 1; y < height - 1; y++)
     {
         if (IsValidExit(0, y)) validExits.Add((0, y));
         if (IsValidExit(width - 1, y)) validExits.Add((width - 1, y));
     }

     // Remove start if it's on the edge
     validExits.RemoveAll(e => e == start);

     // Pick a random valid exit
     if (validExits.Count == 0)
     {
         // fallback: just pick a random edge cell (should be rare)
         exit = (0, 0);
     }
     else
     {
         exit = validExits[rand.Next(validExits.Count)];
     }

     maze[start.y, start.x] = 'S';
     maze[exit.y, exit.x] = 'E';
     player = start;
 }
 ```

## Random number guessing game

A small game that was made as the last assignment for an intro to C# class

This is the entirety of the code for it
```C#
 internal class Program
 {
     static void Main(string[] args)
     {
         const int MIN = 1, MAX = 100;
         int secretNumber;
         int guess;
         int guessAmount = 0;
         bool success;

         Random randomSource = new Random();

         secretNumber = randomSource.Next(MIN, MAX + 1);

         WriteLine("Guess a number between 1 and 100.");

         do
         {
             success = int.TryParse(ReadLine(), out guess);

             while (success == false || guess <= 0 || guess > 100)
             {
                 WriteLine("Error! The guess inputed was not between 1 and 100, please try again.");

                 success = int.TryParse(ReadLine(), out guess);
             }

             guessAmount++;

             if (guess < secretNumber)
             {
                 WriteLine("You have guessed to LOW, please try again.");
             }
             else if (guess > secretNumber)
             {
                 WriteLine("You have guessed to HIGH, please try again");
             }
             else
             {
                 WriteLine("Congratulations, you have guessed the secret number in " + guessAmount + " guesses");
             }

         } while (guess != secretNumber);

         WriteLine("Have a nice day!");
     }
 }
```
